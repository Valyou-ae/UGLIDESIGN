## ğŸ¯ Step 6: OptionChips Component

**File: `src/components/chat/OptionChips.tsx`**
```typescript
import React, { useState } from 'react';
import { OptionChip } from '@/types/chat.types';

interface OptionChipsProps {
  options: OptionChip[];
  onSelect: (value: string) => void;
}

export const OptionChips: React.FC<OptionChipsProps> = ({
  options,
  onSelect
}) => {
  const [showCustomInput, setShowCustomInput] = useState(false);
  const [customValue, setCustomValue] = useState('');
  const [selectedChip, setSelectedChip] = useState<string | null>(null);

  const handleChipClick = (option: OptionChip) => {
    if (option.type === 'custom') {
      setShowCustomInput(true);
    } else {
      setSelectedChip(option.value);
      onSelect(option.value);
    }
  };

  const handleCustomSubmit = () => {
    if (customValue.trim()) {
      onSelect(customValue);
      setCustomValue('');
      setShowCustomInput(false);
    }
  };

  return (
    <div className="option-chips-container">
      {!showCustomInput ? (
        <div className="chips-grid">
          {options.map((option, index) => (
            <button
              key={index}
              className={`option-chip ${option.type === 'custom' ? 'custom-chip' : ''} ${selectedChip === option.value ? 'selected' : ''}`}
              onClick={() => handleChipClick(option)}
              disabled={selectedChip !== null}
            >
              {option.icon && (
                <span className="chip-icon">{option.icon}</span>
              )}
              <span className="chip-label">{option.label}</span>
            </button>
          ))}
        </div>
      ) : (
        <div className="custom-input-container">
          <input
            type="text"
            value={customValue}
            onChange={(e) => setCustomValue(e.target.value)}
            onKeyPress={(e) => e.key === 'Enter' && handleCustomSubmit()}
            placeholder="Type your response..."
            className="custom-input"
            autoFocus
          />
          <button 
            onClick={handleCustomSubmit}
            className="custom-submit-btn"
            disabled={!customValue.trim()}
          >
            Send
          </button>
          <button 
            onClick={() => setShowCustomInput(false)}
            className="custom-cancel-btn"
          >
            Cancel
          </button>
        </div>
      )}
    </div>
  );
};
```

```typescript
import React, { useState, KeyboardEvent } from 'react';

interface MessageInputProps {
  onSend: (message: string) => void;
  disabled?: boolean;
  placeholder?: string;
}

export const MessageInput: React.FC<MessageInputProps> = ({
  onSend,
  disabled = false,
  placeholder = 'Type your message...'
}) => {
  const [input, setInput] = useState('');

  const handleSend = () => {
    if (input.trim() && !disabled) {
      onSend(input.trim());
      setInput('');
    }
  };

  const handleKeyPress = (e: KeyboardEvent<HTMLTextAreaElement>) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  return (
    <div className="message-input-container">
      <textarea
        value={input}
        onChange={(e) => setInput(e.target.value)}
        onKeyPress={handleKeyPress}
        placeholder={placeholder}
        disabled={disabled}
        className="message-input"
        rows={1}
      />
      <button
        onClick={handleSend}
        disabled={disabled || !input.trim()}
        className="send-button"
      >
        <span>â¤</span>
      </button>
    </div>
  );
};
```
